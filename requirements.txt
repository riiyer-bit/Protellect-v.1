from __future__ import annotations
"""
Protellect v2 — AI protein triage platform
Run: streamlit run app.py
"""

import streamlit as st
from styles import inject_css

st.set_page_config(
    page_title="Protellect",
    page_icon="🧬",
    layout="wide",
    initial_sidebar_state="expanded",
)

inject_css()

# ── imports after page config ─────────────────────────────────────────────────
import api, tab_triage, tab_case_study, tab_explorer, tab_experiments

# ── Session state ─────────────────────────────────────────────────────────────
for k, v in {
    "pdata": None, "cv": None, "pdb": "", "papers": [],
    "scored": [], "gene": "", "uid": "", "assay": "", "last": "",
}.items():
    if k not in st.session_state:
        st.session_state[k] = v

# ── Header ────────────────────────────────────────────────────────────────────
st.markdown("""
<div class="proto-header">
  <div class="proto-title">🧬 Protellect</div>
  <div class="proto-sub">AI-powered protein triage · Eliminate wasted experiments · Follow the science</div>
</div>
""", unsafe_allow_html=True)

# ── Sidebar ───────────────────────────────────────────────────────────────────
with st.sidebar:
    st.markdown("""
    <div style='text-align:center;padding:0.5rem 0 0.8rem;'>
      <div style='font-size:2rem;'>🧬</div>
      <div style='color:#00e5ff;font-size:1.3rem;font-weight:800;letter-spacing:-0.5px;'>Protellect</div>
      <div style='color:#1a4a6a;font-size:0.72rem;'>Protein Intelligence Platform</div>
    </div>
    <div style='border-top:1px solid #0d2545;margin-bottom:1rem;'></div>
    """, unsafe_allow_html=True)

    tutorial = st.toggle("📖 Tutorial Mode", value=False)

    st.markdown("<div class='sb-title'>🔍 Protein Search</div>", unsafe_allow_html=True)
    query  = st.text_input("Gene / Protein / UniProt ID", placeholder="TP53 · BRCA1 · P04637 · EGFR")
    search = st.button("🔬 Analyse Protein", use_container_width=True)

    st.markdown("<div class='sb-title'>🧫 Wet-Lab Assay (optional)</div>", unsafe_allow_html=True)
    assay_text = st.text_area("Paste assay description / results", height=100,
                               placeholder="e.g. Western blot shows reduced expression in HEK293T upon CRISPR knockout…")

    st.markdown("<div class='sb-title'>⚙️ Data Depth</div>", unsafe_allow_html=True)
    depth = st.selectbox("ClinVar fetch size", ["Standard (50 variants)", "Deep (300 variants)"], index=0)
    max_v = 50 if "Standard" in depth else 300

    # Sidebar summary (shown after data loads)
    if st.session_state["pdata"]:
        pdata = st.session_state["pdata"]
        gene  = st.session_state["gene"]
        uid   = st.session_state["uid"]
        fname = api.get_protein_name(pdata)
        cv    = st.session_state["cv"]
        scored = st.session_state["scored"]

        st.markdown(f"""
        <div style='border-top:1px solid #0d2545;margin:0.8rem 0 0.5rem;'></div>
        <div class='sb-card'>
          <div style='color:#00e5ff;font-weight:700;font-size:0.95rem;'>{gene}</div>
          <div style='color:#3a6080;font-size:0.78rem;margin:2px 0;'>{fname[:50]}</div>
          <div style='color:#1a3a5a;font-size:0.72rem;'>UniProt: {uid}</div>
        </div>
        """, unsafe_allow_html=True)

        # Disease ranking
        diseases = api.get_diseases(pdata)
        cv_conds = {c for v in (cv or {}).get("variants",[]) for c in v.get("condition","").split(";") if c.strip() != "Not specified"}
        ds_scores = {}
        for v in scored:
            for c in v.get("condition","").split(";"):
                c = c.strip()
                if c: ds_scores[c] = max(ds_scores.get(c,0), v.get("ml",0))

        all_names = [d["name"] for d in diseases] + list(cv_conds)
        all_names  = list(dict.fromkeys(all_names))[:12]

        if all_names:
            st.markdown("<div class='sb-title'>🏥 Disease Affiliations</div>", unsafe_allow_html=True)
            for name in all_names[:10]:
                score = ds_scores.get(name, 0.4)
                rank  = "CRITICAL" if score>=0.85 else "HIGH" if score>=0.65 else "MEDIUM" if score>=0.40 else "NEUTRAL"
                kw = name.lower()
                if any(k in kw for k in ["cancer","carcinoma","leukemia","sarcoma","glioma"]) and rank=="MEDIUM":
                    rank = "HIGH"
                rc = {"CRITICAL":"#ff2d55","HIGH":"#ff8c42","MEDIUM":"#ffd60a","NEUTRAL":"#2a5070"}
                clr = rc[rank]
                st.markdown(
                    f"<div style='display:flex;align-items:center;gap:7px;margin:4px 0;'>"
                    f"<span style='background:rgba(0,0,0,0.4);color:{clr};border:1px solid {clr}55;"
                    f"padding:1px 8px;border-radius:10px;font-size:0.68rem;font-weight:800;"
                    f"white-space:nowrap;'>{rank}</span>"
                    f"<span style='color:#4a7090;font-size:0.78rem;'>{name[:35]}</span>"
                    f"</div>", unsafe_allow_html=True)

        # Assay interpretation
        if assay_text and assay_text.strip():
            tl = assay_text.lower()
            findings = []
            kw_map = [
                (["western","wb","blot"],   "Expression change (WB)"),
                (["crispr","knockout","cas9"],"Loss-of-function model"),
                (["overexpression"],        "Gain-of-function model"),
                (["binding","co-ip"],       "Protein interaction data"),
                (["phospho"],               "PTM/phosphorylation detected"),
                (["apoptosis","caspase"],   "Apoptotic pathway active"),
                (["flow","facs"],           "Flow cytometry data"),
                (["in vivo","xenograft"],   "In vivo model data"),
            ]
            for kws, label in kw_map:
                if any(k in tl for k in kws): findings.append(label)
            if findings:
                st.markdown("<div class='sb-title'>🧫 Assay Findings</div>", unsafe_allow_html=True)
                for f in findings:
                    st.markdown(
                        f"<div style='color:#3a8060;font-size:0.79rem;margin:3px 0;'>"
                        f"✓ {f}</div>", unsafe_allow_html=True)

        # Quick experiment suggestions
        kws_all = [k.get("value","").lower() for k in pdata.get("keywords",[])]
        ptype = ("kinase" if any("kinase" in k for k in kws_all) else
                 "gpcr"  if any("gpcr" in k or "g protein" in k for k in kws_all) else
                 "transcription_factor" if any("transcription" in k for k in kws_all) else "general")
        suggestions = {
            "kinase":["ADP-Glo kinase assay","Phospho-proteomics (MS)","Kinase inhibitor screen","Substrate identification"],
            "gpcr":["cAMP accumulation (HTRF)","β-arrestin recruitment (BRET)","Radioligand binding","Biased agonism profile"],
            "transcription_factor":["ChIP-seq","EMSA (DNA binding)","Luciferase reporter","RNA-seq post-KD"],
            "general":["Co-IP / AP-MS","Subcellular imaging (IF)","CRISPR KO screen","Thermal shift assay"],
        }.get(ptype,["Co-IP / AP-MS","CRISPR KO screen"])
        st.markdown("<div class='sb-title'>🔭 Suggested Experiments</div>", unsafe_allow_html=True)
        for s in suggestions:
            st.markdown(f"<div style='color:#2a5070;font-size:0.78rem;margin:3px 0;'>▸ {s}</div>", unsafe_allow_html=True)

# ── Tutorial banner ───────────────────────────────────────────────────────────
if tutorial:
    st.markdown("""
    <div style='background:#04100a;border:1px solid #00c89633;border-radius:10px;padding:0.9rem 1.2rem;margin-bottom:1rem;'>
      <span style='color:#00c896;font-weight:700;'>📖 Tutorial Mode</span>
      <span style='color:#2a6050;font-size:0.86rem;margin-left:10px;'>
        Step 1: Enter a gene (try <b>TP53</b>) → Click Analyse Protein →
        Step 2: Triage tab = 3D structure & disease ranking →
        Step 3: Case Study = tissue, genomics, GPCR →
        Step 4: Explorer = click residues, simulate mutations →
        Step 5: Experiments = pick cost-effective validated pathways
      </span>
    </div>
    """, unsafe_allow_html=True)

# ── Data loading ──────────────────────────────────────────────────────────────
if search and query and query != st.session_state["last"]:
    with st.spinner("🔬 Fetching from UniProt, ClinVar, AlphaFold & PubMed…"):
        try:
            pdata = api.fetch_uniprot(query)
            st.session_state["pdata"] = pdata
            gene  = api.get_gene_name(pdata)
            uid   = pdata.get("primaryAccession","")
            st.session_state["gene"] = gene
            st.session_state["uid"]  = uid

            cv = api.fetch_clinvar(gene, max_v)
            st.session_state["cv"] = cv

            pdb = api.fetch_alphafold_pdb(uid)
            st.session_state["pdb"] = pdb

            papers = api.fetch_papers(gene)
            st.session_state["papers"] = papers

            scored = api.ml_score_variants(cv.get("variants",[]))
            st.session_state["scored"] = scored

            st.session_state["assay"]  = assay_text
            st.session_state["last"]   = query
            st.rerun()

        except Exception as e:
            st.error(f"⚠️ {e}")

# ── Main content ──────────────────────────────────────────────────────────────
if not st.session_state["pdata"]:
    st.markdown("""
    <div class="info-banner">
      <div style='font-size:3rem;margin-bottom:0.8rem;'>🧬</div>
      <h2>Enter a protein in the sidebar to begin</h2>
      <p>Try: <b>TP53</b> · <b>BRCA1</b> · <b>EGFR</b> · <b>KRAS</b> · <b>P04637</b></p>
      <br>
      <div style='display:flex;gap:1rem;justify-content:center;flex-wrap:wrap;margin-top:0.5rem;'>
        <div style='background:#06101e;border:1px solid #0d2545;border-radius:10px;padding:0.8rem 1.2rem;width:180px;'>
          <div style='color:#00e5ff;font-size:1.3rem;'>🔴</div>
          <div style='color:#4a7fa5;font-size:0.82rem;margin-top:4px;'><b>Triage</b><br>3D structure + ClinVar hotspots</div>
        </div>
        <div style='background:#06101e;border:1px solid #0d2545;border-radius:10px;padding:0.8rem 1.2rem;width:180px;'>
          <div style='color:#00e5ff;font-size:1.3rem;'>📋</div>
          <div style='color:#4a7fa5;font-size:0.82rem;margin-top:4px;'><b>Case Study</b><br>Tissue, genomics, GPCR</div>
        </div>
        <div style='background:#06101e;border:1px solid #0d2545;border-radius:10px;padding:0.8rem 1.2rem;width:180px;'>
          <div style='color:#00e5ff;font-size:1.3rem;'>🔬</div>
          <div style='color:#4a7fa5;font-size:0.82rem;margin-top:4px;'><b>Explorer</b><br>Click residues, mutate</div>
        </div>
        <div style='background:#06101e;border:1px solid #0d2545;border-radius:10px;padding:0.8rem 1.2rem;width:180px;'>
          <div style='color:#00e5ff;font-size:1.3rem;'>🧪</div>
          <div style='color:#4a7fa5;font-size:0.82rem;margin-top:4px;'><b>Experiments</b><br>Protocols + cost guide</div>
        </div>
      </div>
    </div>
    """, unsafe_allow_html=True)

else:
    pdata  = st.session_state["pdata"]
    cv     = st.session_state["cv"]
    pdb    = st.session_state["pdb"]
    papers = st.session_state["papers"]
    scored = st.session_state["scored"]
    gene   = st.session_state["gene"]
    assay  = st.session_state["assay"]

    t1, t2, t3, t4 = st.tabs([
        "🔴 Triage",
        "📋 Case Study",
        "🔬 Protein Explorer",
        "🧪 Experiments & Therapy",
    ])
    with t1: tab_triage.render(pdata, cv, scored, papers)
    with t2: tab_case_study.render(pdata, cv, papers, gene)
    with t3: tab_explorer.render(pdata, pdb, cv, scored, papers, gene)
    with t4: tab_experiments.render(pdata, cv, scored, assay, papers, gene)

# ── Footer ────────────────────────────────────────────────────────────────────
st.markdown(
    "<div class='divider'></div>"
    "<p style='text-align:center;color:#0d2545;font-size:0.74rem;'>"
    "Protellect · Data: UniProt · ClinVar · AlphaFold DB · PubMed · NCBI Gene · "
    "Not a substitute for expert clinical or scientific judgment.</p>",
    unsafe_allow_html=True,
)
